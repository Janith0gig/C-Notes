#include <iostream>
using namespace std;

class Car {
public:
    string make;
    string model;
    int year;

//default constructor
 Car(){
    make="McLaren"; 
    model="p1";
    year= 2014;
 }

    void displayinfo ()
    {
        cout<<year<<" "<<make<<" "<<model<<endl;
    }
};
int main(){
    //creating an object using default constructor
    Car myCar;
    
    //displaying the information
    myCar.displayinfo();
    
    

    
    return 0;
}
